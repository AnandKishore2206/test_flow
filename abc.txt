// scroll-lock.directive.ts
import { Directive, OnInit, OnDestroy } from '@angular/core';
import { OverlayContainer } from '@angular/cdk/overlay';

@Directive({
  selector: '[appScrollLock]'
})
export class ScrollLockDirective implements OnInit, OnDestroy {
  constructor(private overlayContainer: OverlayContainer) {}

  ngOnInit(): void {
    this.overlayContainer.getContainerElement().classList.add('scroll-lock');
  }

  ngOnDestroy(): void {
    this.overlayContainer.getContainerElement().classList.remove('scroll-lock');
  }
}


/* styles.scss */
.scroll-lock {
  overflow: hidden;
  height: 100vh;
}


<!-- flyin.component.html -->
<div class="flyin-wrapper" appScrollLock>
  <!-- Fly-in content -->
</div>


// your.module.ts
import { ScrollLockDirective } from './directives/scroll-lock.directive';

  declarations: [
    ScrollLockDirective,
    // other components
  ]
})
export class YourModule {}
